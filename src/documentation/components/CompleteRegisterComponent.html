    });

      this.ecosystems = data;

    this.completeRegisterService.getContent("life")

      this.focusItems = data;
      console.log(this.focusItems);
      this.contentLoaded=true;

    .then(data=>{
      this.employee = data[0];
      console.log(this.employee);
      //this.employee.admin == yes
      if(this.employee.idorganization != null){
        if(this.employee.idorganization.adminId.email != this.employee.idperson.email){
          this.disableForms();
        }
        this.buttonLabel = "Actualizate";
        this.newActor = true;
        if(this.employee.idorganization.type =="1"){
          this.isCompany = true;
        }
        else if(this.employee.idorganization.type =="2"){
          this.isInvestor = true;
        }
        else if(this.employee.idorganization.type =="3"){
          this.isRdGroup = true;
        }
        else if(this.employee.idorganization.type =="4"){
          this.isStartup = true;
        }
        this.loadActor();
      }
    })

      if(this.goals.commercialization == true)
        this.goals.commercialization = false;
      else{
        this.goals.commercialization = true;
      }
    }

      if(this.goals.industrialCollab == true)
        this.goals.industrialCollab = false;
      else{
        this.goals.industrialCollab = true;
      }
    }

      else{
        this.goals.RDCollab = true;

      if(this.goals.strategicInvestment == true)
        this.goals.strategicInvestment = false;
      else{
        this.goals.strategicInvestment = true;
      }
    }
    if(value == "techAcceleration"){
      if(this.goals.techAcceleration == true)
        this.goals.techAcceleration = false;
      else{
        this.goals.techAcceleration = true;
      }

      if(this.goals.techLicensing == true)

      else{
        this.goals.techLicensing = true;
      }
    }
    
  }

        this.goals.strategicInvestment = true;

        this.goals.techAcceleration = true;

        this.goals.techLicensing = true;

      this.general.size = this.employee.idorganization.companyId.size;
      this.general.establishmentYear = this.employee.idorganization.companyId.establishmentYear;

      for ( let item in this.focusItems){
        if(this.focusString.includes(this.focusItems[item].value)){
          this.focus[item]=true;
        }
      }
      if(this.employee.idorganization.companyId.ecosystem != null && this.employee.idorganization.companyId.technology != null){
        if(this.employee.idorganization.companyId.ecosystem.includes("energy")){
          if(this.employee.idorganization.companyId.technology.includes("automotive")){

            this.energy[4]=true;
          }

            this.energy[5]=true;
          }
        }
        if(this.employee.idorganization.companyId.ecosystem.includes("materials")){
          console.log("materiales");
